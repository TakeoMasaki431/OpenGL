#include <GL/glew.h>
#include <GL/freeglut.h>
#include <iostream>

float posX = 0.0f, posY = 0.0f;
float angle = 0.0f;
float scale = 1.0f;

void drawText(const char* text, float x, float y) {
    glRasterPos2f(x, y);
    while (*text) {
        glutBitmapCharacter(GLUT_BITMAP_HELVETICA_18,
        *text++);
    }
}


void drawHUD() {
    glMatrixMode(GL_PROJECTION);
    glPushMatrix();
    glLoadIdentity();
    gluOrtho2D(0, 1, 0, 1);
    glMatrixMode(GL_MODELVIEW);
    glPushMatrix();
    glLoadIdentity();


    glEnable(GL_BLEND);
    glBlendFunc(GL_SRC_ALPHA, GL_ONE_MINUS_SRC_ALPHA);
    glColor4f(0, 0, 0, 0.6f);

    glBegin(GL_QUADS);
        glVertex2f(0.05f, 0.85f);
        glVertex2f(0.35f, 0.85f);
        glVertex2f(0.35f, 0.65f);
        glVertex2f(0.05f, 0.65f);
    glEnd();

    glDisable(GL_BLEND);
    glColor3f(1, 1, 1);

 
    drawText("Controls:", 0.07f, 0.82f);
    drawText("WASD - Move", 0.07f, 0.78f);
    drawText("Arrow Keys - Rotate", 0.07f, 0.74f);
    drawText("Z/X - Scale", 0.07f, 0.70f);
    drawText("Mouse - Move", 0.07f, 0.66f);

 
    glMatrixMode(GL_PROJECTION);
    glPopMatrix();
    glMatrixMode(GL_MODELVIEW);
    glPopMatrix();
}

void drawSquare() {
    glBegin(GL_QUADS);
        glVertex2f(-0.1f, -0.1f);
        glVertex2f(0.1f, -0.1f);
        glVertex2f(0.1f, 0.1f);
        glVertex2f(-0.1f, 0.1f);
    glEnd();
}

void display() {
    glClear(GL_COLOR_BUFFER_BIT);
    glLoadIdentity();

    glTranslatef(posX, posY, 0);
    glRotatef(angle, 0, 0, 1);
    glScalef(scale, scale, 1);

    glColor3f(0, 1, 1);
    drawSquare();

    drawHUD();

    glutSwapBuffers();
}

void keyboard(unsigned char key, int, int) {
    switch (key) {
        case 'w': posY += 0.05f; break;
        case 's': posY -= 0.05f; break;
        case 'a': posX -= 0.05f; break;
        case 'd': posX += 0.05f; break;
        case 'z': scale += 0.1f; break;
        case 'x': scale -= 0.1f; break;
        case 27: exit(0);
    }
    glutPostRedisplay();
}

void specialKeys(int key, int, int) {
    switch (key) {
        case GLUT_KEY_LEFT: angle += 5.0f; break;
        case GLUT_KEY_RIGHT: angle -= 5.0f; break;
    }
    glutPostRedisplay();
}

void mouseMotion(int x, int y) {
    int width = glutGet(GLUT_WINDOW_WIDTH);
    int height = glutGet(GLUT_WINDOW_HEIGHT);
    posX = (x - width / 2) / (float)(width / 2);
    posY = -(y - height / 2) / (float)(height / 2);
    glutPostRedisplay();
}

int main(int argc, char** argv) {
    glutInit(&argc, argv);
    glutInitDisplayMode(GLUT_DOUBLE | GLUT_RGB);
    glutInitWindowSize(600, 600);
    glutCreateWindow("2D Transformations with HUD");

    glewInit();
    
    glutDisplayFunc(display);
    glutKeyboardFunc(keyboard);
    glutSpecialFunc(specialKeys);
    glutPassiveMotionFunc(mouseMotion);

    glClearColor(0, 0, 0, 1);
    glutMainLoop();
    return 0;
}

#include <GL/glew.h>
#include <GL/freeglut.h>
#include <cmath>
#include <iostream>

float posX = 0.0f, posY = 0.0f; // Θέση
float rotation = 0.0f; // Περιστροφή
float scale = 1.0f; // Κλίμακα
float colorR = 1.0f, colorG = 1.0f, colorB = 1.0f; // Χρώμα
bool autoRotate = false; // Αυτόματη περιστροφή
int shape = 1; // 1: Τετράγωνο, 2: Τρίγωνο, 3: Κύκλος

void drawSquare() {
    glBegin(GL_QUADS);
        glVertex2f(-0.5f, -0.5f);
        glVertex2f(0.5f, -0.5f);
        glVertex2f(0.5f, 0.5f);
        glVertex2f(-0.5f, 0.5f);
    glEnd();
}

void drawTriangle() {
    glBegin(GL_TRIANGLES);
        glVertex2f(0.0f, 0.6f);
        glVertex2f(-0.5f, -0.5f);
        glVertex2f(0.5f, -0.5f);
    glEnd();
}

void drawCircle() {
    glBegin(GL_TRIANGLE_FAN);
    glVertex2f(0.0f, 0.0f);
    for (int i = 0; i <= 360; i += 10) {
        float theta = i * 3.14159f / 180.0f;
        glVertex2f(cos(theta) * 0.5f, sin(theta) * 0.5f);
    }
    glEnd();
}

void display() {
    glClear(GL_COLOR_BUFFER_BIT);
    glLoadIdentity();

    glTranslatef(posX, posY, 0);
    glRotatef(rotation, 0, 0, 1);
    glScalef(scale, scale, 1);

    glColor3f(colorR, colorG, colorB);

    switch (shape) {
        case 1: drawSquare(); break;
        case 2: drawTriangle(); break;
        case 3: drawCircle(); break;
    }

    glutSwapBuffers();
}

void keyboard(unsigned char key, int, int) {
    switch (key) {
        case 'w': posY += 0.1f; break;
        case 's': posY -= 0.1f; break;
        case 'a': posX -= 0.1f; break;
        case 'd': posX += 0.1f; break;
        case 'z': scale += 0.1f; break;
        case 'x': scale -= 0.1f; break;
        case 'c': // Τυχαίο χρώμα
            colorR = static_cast<float>(rand()) / RAND_MAX;
            colorG = static_cast<float>(rand()) / RAND_MAX;
            colorB = static_cast<float>(rand()) / RAND_MAX;
            break;
        case '1': shape = 1; break;
        case '2': shape = 2; break;
        case '3': shape = 3; break;
        case ' ': autoRotate = !autoRotate; break;
        case 27: exit(0);
    }
    glutPostRedisplay();
}

void specialKeys(int key, int, int) {
    switch (key) {
        case GLUT_KEY_LEFT: rotation += 5.0f; break;
        case GLUT_KEY_RIGHT: rotation -= 5.0f; break;
    }
    glutPostRedisplay();
}

void update(int value) {
    if (autoRotate) {
        rotation += 1.0f;
        if (rotation > 360.0f) rotation -= 360.0f;
    }
    glutPostRedisplay();
    glutTimerFunc(16, update, 0);
}

int main(int argc, char** argv) {
    glutInit(&argc, argv);
    glutInitDisplayMode(GLUT_DOUBLE | GLUT_RGB);
    glutInitWindowSize(600, 600);
    glutCreateWindow("2D Μετασχηματισμοί");

    glewInit();
    glClearColor(0, 0, 0, 1);

    glutDisplayFunc(display);
    glutKeyboardFunc(keyboard);
    glutSpecialFunc(specialKeys);
    glutTimerFunc(16, update, 0);

    glutMainLoop();
    return 0;
}

#include <vector>

// Game parameters
const int GRID_SIZE = 20;
int direction = 0;  
// 0: up, 1: down, 2: left, 3: right
bool gameOver = false;
int score = 0;  // Game score

// Structure for snake segments
struct Point {
    int x, y;
};

std::vector<Point> snake =
 { {GRID_SIZE / 2, GRID_SIZE / 2} };
Point food;

// Generates new food at a random position
void generateFood() {
    food.x = rand() % GRID_SIZE;
    food.y = rand() % GRID_SIZE;
}

// Draws a square at position (x, y)
void drawSquare(int x, int y) {
    float size = 2.0f / GRID_SIZE;
    float px = -1.0f + x * size;
    float py = 1.0f - y * size - size;

    glBegin(GL_QUADS);
    glVertex2f(px, py);
    glVertex2f(px + size, py);
    glVertex2f(px + size, py + size);
    glVertex2f(px, py + size);
    glEnd();
}

// Draws the snake and the food
void display() {
    glClear(GL_COLOR_BUFFER_BIT);

    if (gameOver) {
        glColor3f(1, 0, 0);  
        // Red when the game is over
        drawSquare(GRID_SIZE / 2, GRID_SIZE / 2);  
		// Draw the last part of the snake 
		//(for game over display)

        // Display "Game Over" message
        glColor3f(1, 1, 1);
        std::string gameOverText = "Game Over!";
        glRasterPos2f(-0.5f, 0.0f);
        for (char c : gameOverText) {
            glutBitmapCharacter(GLUT_BITMAP_HELVETICA_18,c);
        }
    }
    else {
        glColor3f(1, 0, 0);  // Red food
        drawSquare(food.x, food.y);

        glColor3f(0, 1, 0);  // Green snake
        for (const auto& segment : snake) {
            drawSquare(segment.x, segment.y);
        }

        // Display score
        glColor3f(1, 1, 1);
        std::stringstream ss;
        ss << "Score: " << score;
        std::string scoreText = ss.str();
        glRasterPos2f(-0.9f, 0.9f);  
        // Score at the top left corner
        for (char c : scoreText) {
            glutBitmapCharacter(GLUT_BITMAP_HELVETICA_18,c);
        }
    }

    glutSwapBuffers();
}

// Updates the position of the snake
// Game logic
void update(int value) {
    if (gameOver) return;

    Point newHead = snake.front();

    // Snake movement
    switch (direction) {
    case 0: newHead.y--; break;  // Up
    case 1: newHead.y++; break;  // Down
    case 2: newHead.x--; break;  // Left
    case 3: newHead.x++; break;  // Right
    }

    // Snake wrapping from one side to the other
    if (newHead.x < 0) newHead.x = GRID_SIZE - 1;  
    // If it goes left, wrap to the right
    if (newHead.x >= GRID_SIZE) newHead.x = 0;  
    // If it goes right, wrap to the left
    if (newHead.y < 0) newHead.y = GRID_SIZE - 1;  
    // If it goes up, wrap to the bottom
    if (newHead.y >= GRID_SIZE) newHead.y = 0;  
    // If it goes down, wrap to the top

    // Check if the snake hits itself
    for (const auto& segment : snake) {
        if (segment.x == newHead.x 
        && segment.y == newHead.y) {
            gameOver = true;
            return;
        }
    }

    // If the new head is on food
    if (newHead.x == food.x && newHead.y == food.y) {
        snake.insert(snake.begin(), newHead);  
        // Grow the snake
        generateFood();
        score++;  // Increase score
    }
    else {
        snake.insert(snake.begin(), newHead);
        snake.pop_back();  // Move the snake
    }

    glutPostRedisplay();
    glutTimerFunc(100, update, 0);  
    // Update the game every 100ms
}

// Keyboard handler
void keyboard(int key, int, int) {
    if (gameOver && key == 'r') {
        // Restart the game
        snake.clear();
        snake.push_back({ GRID_SIZE / 2, GRID_SIZE / 2 });
        generateFood();
        score = 0;
        gameOver = false;
        direction = 0;
        glutPostRedisplay();
        return;
    }

    if (key == GLUT_KEY_UP && direction != 1) 
    direction = 0;
    if (key == GLUT_KEY_DOWN && direction != 0) 
    direction = 1;
    if (key == GLUT_KEY_LEFT && direction != 3) 
    direction = 2;
    if (key == GLUT_KEY_RIGHT && direction != 2) 
    direction = 3;
}

// Main function
int main(int argc, char** argv) {
    srand(time(0));
    generateFood();

    glutInit(&argc, argv);
    glutInitDisplayMode(GLUT_DOUBLE | GLUT_RGB);
    glutInitWindowSize(500, 500);
    glutCreateWindow("Snake Game");

    glClearColor(0, 0, 0, 1);
    glutDisplayFunc(display);

#include <GL/glew.h>
#include <GL/freeglut.h>
#include <vector>
#include <cstdlib>
#include <ctime>
#include <sstream>

// Game parameters
const int GRID_SIZE = 20;
const int FOOD_COUNT = 2;  // Number of food items
int direction = 3;  // 0: up, 1: down, 2: left, 3: right
bool gameOver = false;
int score = 0;

// Structure for snake segments and food
struct Point {
    int x, y;
};

std::vector<Point> snake = { {GRID_SIZE / 2, GRID_SIZE / 2} };
std::vector<Point> foodList;  // List of food positions

// Generate food in random positions
void generateFood() {
    foodList.clear();
    for (int i = 0; i < FOOD_COUNT; i++) {
        Point food;
        food.x = rand() % GRID_SIZE;
        food.y = rand() % GRID_SIZE;
        foodList.push_back(food);
    }
}

// Draw a rounded square at (x, y)
void drawRoundedSquare(int x, int y) {
    float size = 2.0f / GRID_SIZE;
    float px = -1.0f + x * size;
    float py = 1.0f - y * size - size;

    glBegin(GL_POLYGON);
    for (int i = 0; i < 360; i += 30) {
        float theta = i * 3.14159f / 180;
        float offsetX = (size / 2) * cos(theta);
        float offsetY = (size / 2) * sin(theta);
        glVertex2f(px + size / 2 + offsetX, py + size / 2 + offsetY);
    }
    glEnd();
}

// Draw the food
void drawFood() {
    glColor3f(1, 0, 0);  // Red food
    for (const auto& food : foodList) {
        drawRoundedSquare(food.x, food.y);
    }
}

// Display function to draw snake, food, and score
void display() {
    glClear(GL_COLOR_BUFFER_BIT);

    if (gameOver) {
        glColor3f(1, 0, 0);
        drawRoundedSquare(snake[0].x, snake[0].y);

        glColor3f(1, 1, 1);
        std::string gameOverText = "Game Over! Press 'R' to Restart.";
        glRasterPos2f(-0.5f, 0.0f);
        for (char c : gameOverText) {
            glutBitmapCharacter(GLUT_BITMAP_HELVETICA_18, c);
        }
    }
    else {
        drawFood();

        glColor3f(0, 1, 0);  // Green snake
        for (const auto& segment : snake) {
            drawRoundedSquare(segment.x, segment.y);
        }

        glColor3f(1, 1, 1);
        std::stringstream ss;
        ss << "Score: " << score;
        std::string scoreText = ss.str();
        glRasterPos2f(-0.9f, 0.9f);
        for (char c : scoreText) {
            glutBitmapCharacter(GLUT_BITMAP_HELVETICA_18, c);
        }
    }

    glutSwapBuffers();
}

// Update function to move the snake
void update(int value) {
    if (gameOver) return;

    Point newHead = snake.front();

    switch (direction) {
    case 0: newHead.y--; break;
    case 1: newHead.y++; break;
    case 2: newHead.x--; break;
    case 3: newHead.x++; break;
    }

    // Wrap-around if snake moves beyond the grid
    if (newHead.x < 0) newHead.x = GRID_SIZE - 1;
    if (newHead.x >= GRID_SIZE) newHead.x = 0;
    if (newHead.y < 0) newHead.y = GRID_SIZE - 1;
    if (newHead.y >= GRID_SIZE) newHead.y = 0;

    // Check if snake collides with itself
    for (const auto& segment : snake) {
        if (segment.x == newHead.x && segment.y == newHead.y) {
            gameOver = true;
            return;
        }
    }

    // Check if snake eats any food
    bool ateFood = false;
    for (size_t i = 0; i < foodList.size(); i++) {
        if (newHead.x == foodList[i].x && newHead.y == foodList[i].y) {
            foodList.erase(foodList.begin() + i);
            ateFood = true;
            break;
        }
    }

    if (ateFood) {
        snake.insert(snake.begin(), newHead);
        generateFood();
        score++;
    }
    else {
        snake.insert(snake.begin(), newHead);
        snake.pop_back();
    }

    glutPostRedisplay();
    glutTimerFunc(100, update, 0);
}

// Keyboard input handling
void keyboard(unsigned char key, int, int) {
    if (gameOver && (key == 'r' || key == 'R')) {
        snake.clear();
        snake.push_back({ GRID_SIZE / 2, GRID_SIZE / 2 });
        generateFood();
        score = 0;
        gameOver = false;
        direction = 3;
        glutPostRedisplay();
        glutTimerFunc(100, update, 0);  // Ξεκινά ξανά το παιχνίδι!
        return;
    }
}

// Arrow key handling
void specialKeyboard(int key, int, int) {
    if (key == GLUT_KEY_UP && direction != 1) direction = 0;
    if (key == GLUT_KEY_DOWN && direction != 0) direction = 1;
    if (key == GLUT_KEY_LEFT && direction != 3) direction = 2;
    if (key == GLUT_KEY_RIGHT && direction != 2) direction = 3;
}

// Main function
int main(int argc, char** argv) {
    srand(time(0));
    generateFood();

    glutInit(&argc, argv);
    glutInitDisplayMode(GLUT_DOUBLE | GLUT_RGB);
    glutInitWindowSize(500, 500);
    glutCreateWindow("Snake Game");

    glClearColor(0, 0, 0, 1);
    glutDisplayFunc(display);
    glutKeyboardFunc(keyboard);
    glutSpecialFunc(specialKeyboard);
    glutTimerFunc(100, update, 0);

    glutMainLoop();
    return 0;
}

Maze Game

#include <GL/glew.h>
#include <GL/freeglut.h>
#include <vector>
#include <cstdlib>
#include <ctime>
#include <iostream>

const int gridSize = 20;
const int cellSize = 20;
const int screenSize = gridSize * cellSize;

struct Point { int x, y; };
std::vector<Point> walls;
Point player = { 0, 0 };
Point exitPoint = { gridSize - 1, gridSize - 1 };
bool gameOver = false;

void drawCell(int x, int y, float r, float g, float b) {
    glColor3f(r, g, b);
    glBegin(GL_QUADS);
    glVertex2f(x * cellSize, y * cellSize);
    glVertex2f((x + 1) * cellSize, y * cellSize);
    glVertex2f((x + 1) * cellSize, (y + 1) * cellSize);
    glVertex2f(x * cellSize, (y + 1) * cellSize);
    glEnd();
}

void drawText(const char* text, int x, int y) {
    glColor3f(1.0, 1.0, 1.0);
    glRasterPos2i(x, y);
    while (*text) {
        glutBitmapCharacter(GLUT_BITMAP_HELVETICA_18, *text);
        text++;
    }
}

void display() {
    glClear(GL_COLOR_BUFFER_BIT);

    for (auto& wall : walls)
        drawCell(wall.x, wall.y, 0.5f, 0.5f, 0.5f);

    drawCell(exitPoint.x, exitPoint.y, 0.0f, 1.0f, 0.0f);
    drawCell(player.x, player.y, 0.0f, 0.0f, 1.0f);

    if (gameOver) {
        drawText("YOU WIN!", screenSize / 2 - 40, screenSize / 2);
    }

    glutSwapBuffers();
}

bool isWall(int x, int y) {
    for (auto& wall : walls)
        if (wall.x == x && wall.y == y)
            return true;
    return false;
}

void addWalls(int value) {
    if (!gameOver) {
        int newX = rand() % gridSize;
        int newY = rand() % gridSize;
        if (!isWall(newX, newY) && !(newX == player.x && newY == player.y) && !(newX == exitPoint.x && newY == exitPoint.y)) {
            walls.push_back({ newX, newY });
        }
        glutPostRedisplay();
        glutTimerFunc(100, addWalls, 0);
    }
}

void keyboard(int key, int, int) {
    if (gameOver) return;

    int newX = player.x, newY = player.y;
    if (key == GLUT_KEY_UP)    newY--;
    if (key == GLUT_KEY_DOWN)  newY++;
    if (key == GLUT_KEY_LEFT)  newX--;
    if (key == GLUT_KEY_RIGHT) newX++;

    if (newX >= 0 && newX < gridSize && newY >= 0 && newY < gridSize && !isWall(newX, newY)) {
        player.x = newX;
        player.y = newY;
    }

    if (player.x == exitPoint.x && player.y == exitPoint.y) {
        gameOver = true;
    }

    glutPostRedisplay();
}

void init() {
    glMatrixMode(GL_PROJECTION);
    glLoadIdentity();
    gluOrtho2D(0, screenSize, screenSize, 0);
    glMatrixMode(GL_MODELVIEW);
    glLoadIdentity();

    srand(time(0));
    glutTimerFunc(1000, addWalls, 0);
}

int main(int argc, char** argv) {
    glutInit(&argc, argv);
    glutInitDisplayMode(GLUT_DOUBLE | GLUT_RGB);
    glutInitWindowSize(screenSize, screenSize);
    glutCreateWindow("Growing Maze Escape");

    glewInit();
    init();

    glutDisplayFunc(display);
    glutSpecialFunc(keyboard);

    glutMainLoop();
    return 0;
}
